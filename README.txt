or
./bin/RiskyBusiness.jar

- ui test
java -cp "bin/RiskyBusiness.jar:lib/*" ui.Main

- rungame
java -cp "bin/RiskyBusiness.jar:lib/*" main.RunGame (On Linux)
java -cp "bin/RiskyBusiness.jar;lib/*" main.RunGame (On Windows)
